<CardGroup cols="2">
  <Card title="1. Design" img="/images/introduction/design.png" href="/editor/fundamentals/artboards" >

  </Card>

  <Card title="2. Animate" img="/images/introduction/animation.gif" href="/editor/animate-mode/animate-mode-overview" >
